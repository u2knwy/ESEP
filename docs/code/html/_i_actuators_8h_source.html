188 void handleEvent(Event event) override {
189 std::lock_guard<std::mutex> lock(mutex);
190
191 bool handled = true;
192
193 switch (event.type)
194 {
195 case EventType::MOTOR_M_STOP:
196 case EventType::MOTOR_S_STOP:
197 motorStop();
198 break;
199 case EventType::MOTOR_M_FAST:
200 case EventType::MOTOR_S_FAST:
201 motorFast();
202 break;
203 case EventType::MOTOR_M_SLOW:
204 case EventType::MOTOR_S_SLOW:
205 motorSlow();
206 break;
207 case EventType::ESTOP_M_PRESSED:
208 case EventType::ESTOP_S_PRESSED:
209 motorStop();
210 break;
211 case EventType::MODE_STANDBY:
212 standbyMode();
213 break;
214 case EventType::MODE_RUNNING:
215 runningMode();
216 break;
217 case EventType::MODE_SERVICE:
218 serviceMode();
219 break;
220 case EventType::MODE_ESTOP:
221 estopMode();
222 break;
223 case EventType::MODE_ERROR:
224 errorMode();
225 break;
226 case EventType::LAMP_M_GREEN:
227 case EventType::LAMP_S_GREEN:
228 case EventType::LAMP_M_YELLOW:
229 case EventType::LAMP_S_YELLOW:
230 case EventType::LAMP_M_RED:
231 case EventType::LAMP_S_RED:
232 case EventType::LED_M_START:
233 case EventType::LED_S_START:
234 case EventType::LED_M_RESET:
235 case EventType::LED_S_RESET:
236 case EventType::LED_M_Q1:
237 case EventType::LED_S_Q1:
238 case EventType::LED_M_Q2:
239 case EventType::LED_S_Q2:
240 handled = handleLampEvent(event.type, (LampState) event.data);
241 break;
242 case EventType::SORT_M_OUT:
243 case EventType::SORT_S_OUT:
244 event.data == 1 ? sortOut() : letPass();
245 break;
246 default:
247 handled = false;
248 }
249
250 if(!handled) {
251 Logger::warn(EVENT_TO_STRING(event.type) + " was not handled by actuators");
252 } else {
253 Logger::debug("[Actuators] Event was handled: " + EVENT_TO_STRING(event.type));
254 }
255 }

257 bool handleLampEvent(EventType event, LampState state) {
258 bool handled = true;
259
260 switch(event) {
261 case LAMP_M_GREEN:
262 case LAMP_S_GREEN:
263 setGreenBlinking(false);
264 if(state == LampState::OFF)
265 greenLampOff();
266 else if(state == LampState::ON)
267 greenLampOn();
268 else if(state == LampState::FLASHING_SLOW)
269 setGreenBlinking(true);
270 else
271 handled = false;
272 break;
273 case LAMP_M_YELLOW:
274 case LAMP_S_YELLOW:
275 setYellowBlinking(false);
276 if(state == LampState::OFF)
277 yellowLampOff();
278 else if(state == LampState::ON)
279 yellowLampOn();
280 else if(state == LampState::FLASHING_SLOW)
281 setYellowBlinking(true);
282 else
283 handled = false;
284 break;
285 case LAMP_M_RED:
286 case LAMP_S_RED:
287 setRedBlinking(false, false);
288 if(state == LampState::OFF)
289 redLampOff();
290 else if(state == LampState::ON)
291 redLampOn();
292 else if(state == LampState::FLASHING_SLOW)
293 setRedBlinking(true, false);
294 else if(state == LampState::FLASHING_FAST)
295 setRedBlinking(true, true);
296 else
297 handled = false;
298 break;
299 case LED_M_START:
300 case LED_S_START:
301 state == LampState::ON ? startLedOn() : startLedOff();
302 break;
303 case LED_M_RESET:
304 case LED_S_RESET:
305 state == LampState::ON ? resetLedOn() : resetLedOff();
306 break;
307 case LED_M_Q1:
308 case LED_S_Q1:
309 state == LampState::ON ? q1LedOn() : q1LedOff();
310 break;
311 case LED_M_Q2:
312 case LED_S_Q2:
313 state == LampState::ON ? q2LedOn() : q2LedOff();
314 break;
315 default:
316 handled = false;
317 }
318
319 return handled;
320 }
